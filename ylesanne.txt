Muudab pealinna Pärnu, kui laps on sündinud 2004.aastal. Tabelis on lapsed.

CREATE TABLE instructor * (
  ID = CHAR(5),
  name = VARCHAR(20) NOT NULL,
  dept_name = VARCHAR(20),
  salary = NUMERIC(8,2),
  PRIMARY KEY = (ID),
  FOREIGN KEY = (dept_name) REFERENCES department(dept_name)
);
